📋 Starter example · copy mine

Want a head start? Copy this whole program — or skip it and build your own flag. draw_rectangle uses a loop for the four sides; draw_circle fills the center button. Whatever you draw last appears on top.

Python code
# Layer 3 · Flag Challenge — rectangles and circles
import turtle

t = turtle.Turtle()
t.speed(0)
t.width(3)

# pen up → jump → pen down
def move_to(x, y):
    t.penup()
    t.goto(x, y)
    t.pendown()

def draw_rectangle(x, y, width, height, color):
    t.color("black", color)
    move_to(x, y)
    t.begin_fill()

    for side in range(2):
        t.forward(width)
        t.left(90)
        t.forward(height)
        t.left(90)

    t.end_fill()

def draw_circle(x, y, radius, color):
    t.color("black", color)
    move_to(x, y - radius)
    t.begin_fill()
    t.circle(radius)
    t.end_fill()

# Background first, details last
draw_rectangle(-250, 0, 500, 140, "red")
draw_rectangle(-250, -140, 500, 140, "white")
draw_rectangle(-250, -15, 500, 30, "black")

draw_circle(0, 0, 55, "black")
draw_circle(0, 0, 35, "white")
draw_circle(0, 0, 12, "lightgray")

t.hideturtle()
turtle.done()
