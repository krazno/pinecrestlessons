Flag Challenge

  1. You
  2. A team

  4. An invented country

  2. Use at least 1 function (preferably with parameters).

Teacher example · code

Simple striped flag with a star — remix the colors and symbols for your own design.

import turtle

t = turtle.Turtle()
t.speed(0)
screen = turtle.Screen()
screen.bgcolor("navy")
t.hideturtle()

def stripe(y, height, color_name):
    """Draw one horizontal stripe on the flag."""
    t.penup()
    t.goto(-200, y)
    t.pendown()
    t.color(color_name)
    t.begin_fill()
    for _ in range(2):
        t.forward(400)
        t.left(90)
        t.forward(height)
        t.left(90)
    t.end_fill()

# stripes in a loop
colors = ["crimson", "white", "crimson"]
for i, c in enumerate(colors):
    stripe(80 - i * 50, 40, c)

def star(size, color_name):
    t.penup()
    t.goto(-120, 20)
    t.pendown()
    t.color(color_name)
    t.begin_fill()
    for _ in range(5):
        t.forward(size)
        t.right(144)
    t.end_fill()

star(40, "gold")  # symbol on the flag

turtle.done()
