Teacher example

Mr. Brett loves space, so this sample draws colorful planets. Pick your own topic and change the code to match your passion.

import turtle

t = turtle.Turtle()
t.speed(0)
screen = turtle.Screen()
screen.bgcolor("midnightblue")

def draw_planet(x, y, size, color_name):
    # One filled circle = one planet
    t.penup()
    t.goto(x, y)
    t.pendown()
    t.color(color_name)
    t.begin_fill()
    t.circle(size)
    t.end_fill()

# Four planets — three colors in the list
planet_colors = ["orangered", "gold", "deepskyblue"]

for i in range(4):
    if i % 2 == 0:
        draw_planet(-140 + i * 75, -20, 28 + i * 6, planet_colors[i % 3])
    else:
        draw_planet(-140 + i * 75, 35, 20 + i * 4, planet_colors[(i + 1) % 3])

turtle.done()
